function renderRelicCards(relics) { const grid = document.getElementById('relic-grid'); grid.innerHTML = ""; selectedRelic = ""; relics.forEach(relic => { const card = document.createElement('div'); card.className = 'relic-card'; const mCount = parseInt(relic.main || 0); if (mCount >= 3) card.classList.add('full'); card.innerHTML = `
${relic.name}
${relic.status || ' '}
👥 ${relic.count}
(${relic.main} / ${relic.sub})
`; card.onclick = () => { const isActive = card.classList.contains('active'); document.querySelectorAll('.relic-card').forEach(c => c.classList.remove('active')); if (!isActive) { card.classList.add('active'); // [수정] 괄호와 상태값을 제외하고 순수 유적 이름만 저장합니다. selectedRelic = relic.name.trim(); card.dataset.mainCount = relic.main; } else { selectedRelic = ""; } }; grid.appendChild(card); }); } async function submitData() { if (!selectedRelic) { alert("신청할 유적을 먼저 클릭해 주세요!"); return; } const historyRows = document.getElementById('history-body').children; // [규칙 반영] 현재 '메인' 신청 기록이 하나도 없는 경우에만 다음 신청이 메인이 됨 let hasMainHistory = false; for(let row of historyRows) { if(row.innerText.includes("메인")) { hasMainHistory = true; break; } } const isNextApplyMain = !hasMainHistory; const activeCard = document.querySelector('.relic-card.active'); const targetMainCount = parseInt(activeCard.dataset.mainCount || 0); if (isNextApplyMain && targetMainCount >= 3) { alert("⚠️ 해당 유적은 메인 신청 인원이 모두 채워져 신청이 불가능합니다."); return; } setLoader(true, "신청 중..."); try { const res = await fetch(GAS_URL, { method: "POST", // [확인] selectedRelic에는 이미 순수 이름만 들어가 있습니다. body: JSON.stringify({ type: "submit", nick: currentUserNick, relic: selectedRelic }) }); const data = await res.json(); if (data.result === "success") { alert("✅ 신청 완료!"); selectedRelic = ""; await login(true); } else { alert("⚠️ " + (data.message || "신청에 실패했습니다.")); } } catch (e) { alert("통신 에러"); } finally { setLoader(false); } }
